# 0.5.7 Event Object Schema

## Overview

Event Objects represent occurrences in time—meetings, transactions, historical events, scheduled activities, and any other time-bound phenomena. They connect people, places, and things through temporal relationships.

## Schema Definition

```yaml
event_object:
  # ═══════════════════════════════════════════════════════════════
  # BASE OBJECT (inherited from 0.5.0)
  # ═══════════════════════════════════════════════════════════════
  identity:
    address: "7.[category].[instance]"
    object_id: "[UUID]"
    object_type: "event"
    subtype: "meeting|transaction|occurrence|scheduled|historical"
    version: "1.0.0"

  # ═══════════════════════════════════════════════════════════════
  # EVENT-SPECIFIC CONTENT
  # ═══════════════════════════════════════════════════════════════
  content:
    # ─────────────────────────────────────────────────────────────
    # EVENT IDENTITY
    # ─────────────────────────────────────────────────────────────
    event_identity:
      title: "Event Title"
      description: "Detailed description of the event"
      short_description: "Brief summary"

      category: "meeting|conference|transaction|milestone|incident|celebration"
      tags: ["tag1", "tag2"]

      images:
        banner: "[Media Object reference]"
        gallery: ["[Media references]"]

    # ─────────────────────────────────────────────────────────────
    # TIMING
    # ─────────────────────────────────────────────────────────────
    timing:
      temporal_type: "instant|duration|recurring|ongoing"

      # For instant events
      instant:
        timestamp: "2025-01-30T14:00:00Z"

      # For duration events
      duration:
        start: "2025-01-30T14:00:00Z"
        end: "2025-01-30T16:00:00Z"
        timezone: "America/Los_Angeles"

        breakdown:
          - segment: "Registration"
            start: "2025-01-30T14:00:00Z"
            end: "2025-01-30T14:15:00Z"
          - segment: "Main Session"
            start: "2025-01-30T14:15:00Z"
            end: "2025-01-30T15:45:00Z"
          - segment: "Q&A"
            start: "2025-01-30T15:45:00Z"
            end: "2025-01-30T16:00:00Z"

      # For recurring events
      recurrence:
        pattern: "daily|weekly|monthly|yearly|custom"
        interval: 1
        days_of_week: ["monday", "wednesday", "friday"]
        start_date: "2025-01-01"
        end_date: "2025-12-31"
        exceptions: ["2025-07-04", "2025-12-25"]

      # Status
      status: "scheduled|in_progress|completed|cancelled|postponed"

    # ─────────────────────────────────────────────────────────────
    # LOCATION
    # ─────────────────────────────────────────────────────────────
    location:
      type: "physical|virtual|hybrid"

      physical_location: "[Location Object reference]"

      virtual_location:
        platform: "hypernet_vr|zoom|teams|custom"
        space: "[Location Object reference]"  # For Hypernet VR
        url: "https://..."  # For external platforms
        access_code: "[Encrypted]"

      hybrid:
        physical: "[Location Object reference]"
        virtual: "[Location Object reference]"
        primary: "physical|virtual"

    # ─────────────────────────────────────────────────────────────
    # PARTICIPANTS
    # ─────────────────────────────────────────────────────────────
    participants:
      organizers:
        - person: "[Person Object reference]"
          role: "host"

      attendees:
        - person: "[Person Object reference]"
          status: "confirmed|tentative|declined|no_response"
          role: "speaker|attendee|moderator"
          attendance:
            registered: "2025-01-20T10:00:00Z"
            checked_in: "2025-01-30T13:55:00Z"
            checked_out: "2025-01-30T16:05:00Z"

      capacity:
        minimum: 5
        maximum: 100
        current_registered: 67
        waitlist: 12

      registration:
        required: true
        open: "2025-01-01T00:00:00Z"
        close: "2025-01-29T23:59:59Z"
        approval_required: false

    # ─────────────────────────────────────────────────────────────
    # CONTENT AND AGENDA
    # ─────────────────────────────────────────────────────────────
    content:
      agenda:
        - time: "14:00"
          item: "Welcome and Introduction"
          presenter: "[Person reference]"
          duration_minutes: 15

        - time: "14:15"
          item: "Main Presentation"
          presenter: "[Person reference]"
          duration_minutes: 60
          materials: "[Document reference]"

        - time: "15:15"
          item: "Break"
          duration_minutes: 15

        - time: "15:30"
          item: "Discussion"
          duration_minutes: 30

      materials:
        pre_event:
          - title: "Pre-reading"
            document: "[Document reference]"
        during_event:
          - title: "Slides"
            document: "[Document reference]"
        post_event:
          - title: "Recording"
            media: "[Media reference]"
          - title: "Notes"
            document: "[Document reference]"

    # ─────────────────────────────────────────────────────────────
    # OUTCOMES (for completed events)
    # ─────────────────────────────────────────────────────────────
    outcomes:
      summary: "Brief summary of what happened"

      decisions:
        - decision: "Decision made during event"
          document: "[Document reference]"

      action_items:
        - task: "[Task Object reference]"
          assigned_to: "[Person reference]"
          due: "2025-02-15"

      recordings:
        - type: "video"
          media: "[Media reference]"
        - type: "transcript"
          document: "[Document reference]"

      attendance_stats:
        registered: 67
        attended: 58
        peak_concurrent: 52
        average_duration_minutes: 95

    # ─────────────────────────────────────────────────────────────
    # FOR TRANSACTION EVENTS
    # ─────────────────────────────────────────────────────────────
    transaction:  # If subtype is "transaction"
      transaction_type: "purchase|transfer|agreement|exchange"

      parties:
        - party: "[Person/Organization reference]"
          role: "buyer|seller|sender|receiver"

      items:
        - item: "[Object reference]"
          quantity: 1
          value: {amount: 100, currency: "USD"}

      total_value: {amount: 100, currency: "USD"}

      status: "pending|completed|cancelled|disputed"

      documents:
        - type: "receipt"
          document: "[Document reference]"
        - type: "contract"
          document: "[Document reference]"

  # ═══════════════════════════════════════════════════════════════
  # LINKS
  # ═══════════════════════════════════════════════════════════════
  links:
    # Organizer
    - link_type: "organized_by"
      target: "[Person/Organization address]"

    # Location
    - link_type: "located_at"
      target: "[Location address]"

    # Participants
    - link_type: "attended_by"
      target: "[Person address]"
      metadata: {role: "speaker"}

    # Part of series
    - link_type: "part_of"
      target: "[Event series address]"

    # Related events
    - link_type: "follows"
      target: "[Previous event address]"

    # Outcomes
    - link_type: "produced"
      target: "[Document/Task address]"
```

## Event Types

### Meetings
Gatherings for discussion, collaboration, or decision-making.

### Transactions
Exchanges of value, ownership transfers, agreements.

### Occurrences
Incidents, accidents, natural events, observations.

### Scheduled
Planned future activities, appointments, deadlines.

### Historical
Past events of significance, documented for the record.
